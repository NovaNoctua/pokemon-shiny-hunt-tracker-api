// 1. Users and Profiles
Table t_user {
  user_id int [pk, increment]
  username varchar(50)
  email varchar(100)
  password varchar(255)
  profile_picture varchar(255)
}

// 2. Pokemon Data (Static)
Table t_pokemon {
  pokemon_id int [pk] // Use National Dex ID
  name varchar(50)
  dex_number int
  generation int
}

// 3. Game Data (Static)
Table t_game {
  game_id int [pk, increment]
  name varchar(50)
  generation int
}

// 4. Methods (Static - e.g. Masuda, Soft Reset)
Table t_method {
  method_id int [pk, increment]
  name varchar(50)
  base_odds varchar(20)
}

// 5. Linking Games to Methods (Many-to-Many resolved)
Table t_methods_in_game {
  game_id int
  method_id int
  
  // Composite Primary Key definition is not strictly needed for the visual, 
  // but logically these two combined are unique.
}

// 6. Active Hunts (What the user is doing NOW)
Table t_hunt {
  hunt_id int [pk, increment]
  started_at date
  current_counter int [default: 0]
  status varchar(20) // 'active', 'paused', etc.
  
  // Foreign Keys (ids only)
  user_id int
  pokemon_id int
  game_id int
  method_id int
}

// 7. Collection (Completed Hunts / History)
Table t_entry {
  entry_id int [pk, increment]
  nickname varchar(50)
  obtained_at date
  final_counter int
  notes text
  is_shiny boolean [default: true]
  phase_number int [default: 1]
  
  // Foreign Keys (ids only)
  user_id int
  pokemon_id int
  game_id int
  method_id int
}

// --- RELATIONSHIPS (The Lines in the Diagram) ---

// Connecting Methods to Games
Ref: t_methods_in_game.game_id > t_game.game_id
Ref: t_methods_in_game.method_id > t_method.method_id

// Connecting Active Hunts
Ref: t_hunt.user_id > t_user.user_id
Ref: t_hunt.pokemon_id > t_pokemon.pokemon_id
Ref: t_hunt.game_id > t_game.game_id
Ref: t_hunt.method_id > t_method.method_id

// Connecting Collection Entries
Ref: t_entry.user_id > t_user.user_id
Ref: t_entry.pokemon_id > t_pokemon.pokemon_id
Ref: t_entry.game_id > t_game.game_id
Ref: t_entry.method_id > t_method.method_id
